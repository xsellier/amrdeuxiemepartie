  \end{itemize}

\begin{frame}
\frametitle{L'algorithme approché en instance critique}
\begin{block}{Un arbre a un seul fils a chaque fois}
  \begin{itemize}
  \item L'algorithme approché va prendre tous les neouds
  \item Le temps de calcul sera faible, mais la quantité de mémoire
    consommée sera doublée par rapport à la solution optimale.
  \end{itemize}
\end{block}
\end{frame}
